PowerShell 測試指令

# 設定 Token（每次開新 PowerShell 視窗需重新設定）
$token = "貼上 LINE_CHANNEL_ACCESS_TOKEN"

# 測試傳送個人通知
$userId = "貼上 LINE_USER_ID"
$body = '{"to":"' + $userId + '","messages":[{"type":"text","text":"測試通知 📊"}]}'
Invoke-RestMethod -Method Post `
  -Uri "https://api.line.me/v2/bot/message/push" `
  -Headers @{"Authorization"="Bearer $token"; "Content-Type"="application/json"} `
  -Body ([System.Text.Encoding]::UTF8.GetBytes($body))

# 測試傳送群組通知
$groupId = "貼上 LINE_GROUP_ID"
$body = '{"to":"' + $groupId + '","messages":[{"type":"text","text":"測試通知 📊"}]}'
Invoke-RestMethod -Method Post `
  -Uri "https://api.line.me/v2/bot/message/push" `
  -Headers @{"Authorization"="Bearer $token"; "Content-Type"="application/json"} `
  -Body ([System.Text.Encoding]::UTF8.GetBytes($body))

# 確認 Bot 資訊（驗證 token 是否有效）
Invoke-RestMethod -Method Get `
  -Uri "https://api.line.me/v2/bot/info" `
  -Headers @{"Authorization"="Bearer $token"}
注意：PowerShell 中不可使用 curl（是 Invoke-WebRequest 的別名），請一律使用 Invoke-RestMethod。
